// JarvisMini working xml differences vs Checkmate broken xml:
//   1. android:label="@string/app_name" — MISSING in Checkmate → ColorOS
//      requires label to show service in Settings → without it the service
//      entry is malformed → bind silently fails
//   2. eventTypes: typeWindowContentChanged|typeNotificationStateChanged
//      (not typeWindowStateChanged — that's why we never got Conversation events)
//   3. notificationTimeout: 50 (not 100)
//   4. description string exists in strings.xml ✓ (confirmed)
//
// Fix: match JarvisMini's xml exactly, add label, fix event types.

//===== FILE: app/src/main/res/xml/accessibility_service.xml =====
<?xml version="1.0" encoding="utf-8"?>
<accessibility-service
    xmlns:android="http://schemas.android.com/apk/res/android"
    android:accessibilityEventTypes="typeWindowContentChanged|typeNotificationStateChanged|typeWindowStateChanged"
    android:packageNames="com.whatsapp,com.checkmate"
    android:accessibilityFeedbackType="feedbackGeneric"
    android:notificationTimeout="50"
    android:canRetrieveWindowContent="true"
    android:description="@string/service_desc"
    android:label="@string/app_name" />


//===== FILE: app/src/main/res/values/strings.xml =====
<resources>
    <string name="app_name">Checkmate</string>
    <string name="service_desc">Checkmate automation service for guardian notifications.</string>
</resources>
