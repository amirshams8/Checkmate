   6     * WorkModeSchedule — the hardcoded daily Block Mode window.
   7     *
   8     * Checkmate enforces Work Mode (app + website blocking, plus the Work Mode
   9     * settings lock in WorkModeLockGate) every day from 19:00 (7 PM) to 05:00
  10     * (5 AM) the following morning — on top of whatever task-based sessions the
  11     * student starts manually.
  12     *
  13     * UPDATE: the usual hardcoded window's end was moved from 02:00 to 05:00
  14     * (still starts 19:00). Sunday and Wednesday additionally get a second,
  15     * earlier lock window — 01:00 to 17:30 — so on those two days the student
  16     * is locked out almost all day with only a short 17:30-19:00 gap before the
  17     * usual 19:00-05:00 window takes back over ("locked twice" per day).
  18     *
  19     * This window is intentionally a compile-time constant. There is no
  20     * Settings screen, SharedPrefs key, or remote config that can change it —
  21     * changing the hours means editing this file and shipping a new build, not
  22     * something reachable from the student's phone. That's deliberate: a
  23     * schedule the student can adjust "just this once" isn't a schedule.
  24     */

  26

  28        const val START_HOUR = 19
  29    
  30        /**
  31         * 24h clock hour the usual daily window ends, on the next calendar day
  32         * (5 AM). Previously 2 AM.
  33         */
  34        const val END_HOUR = 5

  37        private val SPECIAL_DAYS = setOf(Calendar.SUNDAY, Calendar.WEDNESDAY)

  53         *  - the usual 19:00-05:00 window, every day, OR
  54         *  - on Sunday/Wednesday only, the extra 01:00-17:30 window.
  55         */
  56        fun isWithinScheduledWindow(cal: Calendar = Calendar.getInstance()): Boolean {
  57            val hour = cal.get(Calendar.HOUR_OF_DAY)
  58    
  59            // Usual window wraps past midnight: active from START_HOUR..23:59 AND
  60            // 00:00..(END_HOUR-1):59. Applies every day of the week.
  61            if (hour >= START_HOUR || hour < END_HOUR) return true
  62    
  63            // Extra Sunday/Wednesday window: 01:00 (inclusive) to 17:30 (exclusive).
  64            // Doesn't wrap past midnight, so it's a plain same-day range check.
  65            val day = cal.get(Calendar.DAY_OF_WEEK)

  67                val minute = cal.get(Calendar.MINUTE)
  68                val afterSpecialStart = hour > SPECIAL_START_HOUR ||
  69                    (hour == SPECIAL_START_HOUR && minute >= SPECIAL_START_MINUTE)
  70                val beforeSpecialEnd = hour < SPECIAL_END_HOUR ||
  71                    (hour == SPECIAL_END_HOUR && minute < SPECIAL_END_MINUTE)
  72                if (afterSpecialStart && beforeSpecialEnd) return true
  73            }
  74    
  75            return false
  76        }

  97                    set(Calendar.SECOND, 0)
  98                    set(Calendar.MILLISECOND, 0)
  99                    set(Calendar.HOUR_OF_DAY, h)
 100                    set(Calendar.MINUTE, if (h == SPECIAL_END_HOUR) SPECIAL_END_MINUTE else 0)
 101                }
 102                if (target.timeInMillis <= now) target.add(Calendar.DAY_OF_YEAR, 1)
