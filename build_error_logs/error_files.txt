 164        /** Depth-first collection of all text + content-description on screen, space-joined. */
 165        private fun collectAllText(root: AccessibilityNodeInfo): String {
 166            val sb = StringBuilder()
 167            val stack = ArrayDeque<AccessibilityNodeInfo>()
 168            stack.add(root)
 169            var visited = 0
 170            while (stack.isNotEmpty() && visited < 400) { // cap traversal — Settings trees are small
 171                val node = stack.removeFirst()
 172                visited++
 173                node.text?.let { sb.append(it).append(' ') }
 174                node.contentDescription?.let { sb.append(it).append(' ') }
 175                for (i in 0 until node.childCount) node.getChild(i)?.let { stack.add(it) }
 176            }
 177            return sb.toString()

 205         * Tries known view IDs for the browser first, then falls back to
 206         * scanning all text nodes whose content-description suggests a URL bar.
 207         */
 208        private fun extractUrl(root: AccessibilityNodeInfo, pkg: String): String? {
 209            // Try known IDs for this browser
 210            val ids = URL_BAR_IDS[pkg] ?: emptyList()
 211            for (id in ids) {
 212                val node = root.findAccessibilityNodeInfosByViewId(id)?.firstOrNull()
 213                val text = node?.text?.toString()?.trim()
 214                if (!text.isNullOrBlank()) return text
 215            }
 216    
 217            // Generic fallback: find any node whose description sounds like an address bar
 218            // and whose text looks like a URL
 219            val stack = ArrayDeque<AccessibilityNodeInfo>()
 220            stack.add(root)
 221            while (stack.isNotEmpty()) {
 222                val node = stack.removeFirst()
 223                val desc = node.contentDescription?.toString() ?: ""
 224                val text = node.text?.toString()?.trim() ?: ""
 225                if (URL_BAR_DESCRIPTIONS.any { desc.contains(it, ignoreCase = true) }
 226                    && looksLikeUrl(text)) {
 227                    return text
 228                }
 229                for (i in 0 until node.childCount) node.getChild(i)?.let { stack.add(it) }
 230            }
 231            return null
 232        }

 234        private fun looksLikeUrl(text: String): Boolean {
 235            if (text.isBlank() || text.length < 4) return false
 236            return text.contains('.') &&
 237                   !text.contains(' ') &&
 238                   (text.startsWith("http") || text.startsWith("www.") || text.matches(Regex(".*\\.[a-z]{2,}.*")))
 239        }

 241        /**
 242         * Parses the registered hostname from a raw URL string.
 243         * "https://www.youtube.com/watch?v=..." → "youtube.com"
 244         *
 245         * Strips "www." so the stored domain "youtube.com" matches both
 246         * "youtube.com" and "www.youtube.com".
 247         */
 248        private fun parseHostname(raw: String): String? {
 249            return try {
 250                val withScheme = if (raw.startsWith("http")) raw else "https://$raw"
 251                val host = Uri.parse(withScheme).host ?: return null
 252                host.removePrefix("www.").lowercase().trim()
 253            } catch (_: Exception) { null }
 254        }

 305        private fun findNodeByClass(root: AccessibilityNodeInfo, className: String): AccessibilityNodeInfo? {
 306            val stack = ArrayDeque<AccessibilityNodeInfo>()
 307            stack.add(root)
 308            while (stack.isNotEmpty()) {
 309                val node = stack.removeFirst()
 310                if (node.className?.toString() == className && node.isEditable) return node
 311                for (i in 0 until node.childCount) node.getChild(i)?.let { stack.add(it) }
 312            }
 313            return null
 314        }

 316        private fun findNodeByDesc(root: AccessibilityNodeInfo, desc: String): AccessibilityNodeInfo? {
 317            val stack = ArrayDeque<AccessibilityNodeInfo>()
 318            stack.add(root)
 319            while (stack.isNotEmpty()) {
 320                val node = stack.removeFirst()
 321                if (node.contentDescription?.toString()?.contains(desc, ignoreCase = true) == true
 322                    && node.isClickable) return node
 323                for (i in 0 until node.childCount) node.getChild(i)?.let { stack.add(it) }
 324            }
 325            return null
